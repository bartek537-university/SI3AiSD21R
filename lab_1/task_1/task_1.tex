    \section{Wyszukiwanie największej liczby w zbiorze.}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\
        Wczytaj zbiór input\_text\\
        input\_text $\gets$ input\_text + ";"\\
        max\_number $\gets$ BRAK\\
        current\_number $\gets$ 0.0\\
        decimal\_numerator $\gets$ 0\\
        is\_negative $\gets$ fałsz\\

        \For{i $\gets$ 1 \KwTo \Length (input\_text)}{
            current\_char $\gets$ input\_text[i]\\

            \If{current\_char jest liczbą}{
                \If{decimal\_numerator $>$ 0}{
                    current\_number $\gets$ current\_number $+$ \ToNumber (current\_char) $*$ 10^{-decimal\_numerator}\\
                    decimal\_numerator $\gets$ decimal\_numerator + 1\\
                }
                \Else{
                    current\_number $\gets$ current\_number $*$ 10 $+$ \ToNumber (current\_char)\\
                }
                \Continue
            }
            \If{current\_char $=$ "."}{
                decimal\_numerator $\gets$ 1\\
                \Continue
            }
            \If{current\_char $=$ "-"}{
                is\_negative $\gets$ prawda\\
                \Continue
            }
            \If{is\_negative}{
                current\_number $\gets$ current\_number * -1\\
            }
            \If{max\_number $=$ BRAK \textbf{lub} current\_number $>$ max\_number}{
                max\_number $\gets$ current\_number\\
            }

            current\_number $\gets$ 0.0\\
            decimal\_numerator $\gets$ 0\\
            is\_negative $\gets$ fałsz\\
        }
        Wypisz \textbf{max\_number}\\
        Stop\\

        \label{alg:algorithm}
    \end{algorithm}
