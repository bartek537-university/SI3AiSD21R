    \section{Zamiana notacji infiksowej w postfiksową.}

    \begin{algorithm}[H]
        \SetAlgoLined

        Start\\

        operation\_order $\gets$ \textbf{słownik}\{"(" $\gets$ 0, "+" $\gets$ 1, "-" $\gets$ 1, "*" $\gets$ 2, "/" $\gets$ 2, "\textasciicircum" $\gets$ 3\}\\

        Wczytaj równanie w notacji inteksowej infix\_equation\\

        values $\gets$ pusty stos typu tekstowego\\
        operations $\gets$ pusty stos typu tekstowego\\
        is\_previous\_symbol\_numeric $\gets$ fałsz\\

        \For{i $\gets$ 1 \KwTo \Length (input\_text)}{
            current\_char $\gets$ input\_text[i]\\

            \If{current\_char jest liczbą}{
                \If{is\_previous\_symbol\_numeric}{
                    temp1 $\gets$ Wartość z góry stosu \textbf{values}\\
                    Usuń wartość z góry stosu \textbf{values}\\
                    temp2 $\gets$ temp1 $+$ current\_char\\
                    Wstaw na górę stosu \textbf{values} temp2\\
                }
                \Else{
                    Wstaw na górę stosu \textbf{values} current\_char\\
                }
                \Continue
            }

            is\_previous\_symbol\_numeric $\gets$ fałsz\\

            \If{current\_char $=$ "\space"} {
                \Continue
            }
            \If{current\_char $=$ "("} {
                Wstaw na górę stosu \textbf{operations} current\_char\\
                \Continue
            }
            \If{current\_char $=$ ")"} {
                \While{wartość z góry stosu \textbf{operations} != "("} {
                    right\_operand $\gets$ Wartość z góry stosu \textbf{values}\\
                    Usuń wartość z góry stosu \textbf{values}\\
                    left\_operand $\gets$ Wartość z góry stosu \textbf{values}\\
                    Usuń wartość z góry stosu \textbf{values}\\
                    operation $\gets$ Wartość z góry stosu \textbf{operations}\\
                    Usuń wartość z góry stosu \textbf{operations}\\
                    temp1 $\gets$ left\_operand $+$ "\space" $+$ operation $+$ "\space" $+$ right\_operand \\
                    Wstaw na górę stosu \textbf{values} temp1\\
                }
                Usuń wartość z góry stosu \textbf{operations}\\
                \Continue
            }
            \While{operation\_order[current\_char] < operation\_order[Wartość z góry stosu \textbf{operations}]} {
                right\_operand $\gets$ Wartość z góry stosu \textbf{values}\\
                Usuń wartość z góry stosu \textbf{values}\\
                left\_operand $\gets$ Wartość z góry stosu \textbf{values}\\
                Usuń wartość z góry stosu \textbf{values}\\
                operation $\gets$ Wartość z góry stosu \textbf{operations}\\
                Usuń wartość z góry stosu \textbf{operations}\\
                temp1 $\gets$ left\_operand $+$ "\space" $+$ operation $+$ "\space" $+$ right\_operand \\
                Wstaw na górę stosu \textbf{values} temp1\\
            }
            Wstaw na górę stosu \textbf{operations} current\_char\\
        }

        Wypisz Wartość z góry stosu \textbf{values}\\
        Stop\\

        \label{alg:algorithm1}
    \end{algorithm}
