    \subsection{Dodawanie elementów ($push$).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        Wczytaj $value$\\

        \If{$size \geq capacity$}{
            Wypisz błąd o przepełnieniu stosu\\
        }
        \Else{
            $values[size] \gets value$\\
            $size \gets size + 1$\\
        }

        Stop\\
        \label{alg:algorithm1}
    \end{algorithm}

    \subsection{Usuwanie elementów ($pop$).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        \If{$size < 1$}{
            Wypisz błąd o pustym stosie.\\
        }
        \Else{
            $popped\_value \gets values[size - 1]$\\
            $size \gets size - 1$\\

            Wypisz $popped\_value$\\
        }

        Stop\\
        \label{alg:algorithm2}
    \end{algorithm}

    \subsection{Sprawdzanie elementu z góry ($peek$).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        \If{$size < 1$}{
            Wypisz błąd o pustym stosie.\\
        }
        \Else{
            Wypisz $values[size - 1]$\\
        }

        Stop\\
        \label{alg:algorithm3}
    \end{algorithm}
