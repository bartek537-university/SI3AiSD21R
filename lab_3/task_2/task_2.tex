    \section{Kolejka priorytetowa}

    \subsection{Dodawanie elementów ($enqueue$).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        Wczytaj $value$ oraz $priority$\\

        \If{$size \geq capacity$}{
            Wypisz błąd o przepełnieniu bufora.\\
        }
        \Else{
            $i \gets size$

            \While{$i > 0$}{
                \If{$priorities[i-1] > priority$}{
                    $values[i] \gets values[i-1]$\\
                    $priorities[i] \gets priorities[i-1]$\\
                    $i \gets i - 1$\\
                }
                \Else{
                    \Break
                }
            }

            $values[i] \gets value$\\
            $priorities[i] \gets priority$\\

            $size \gets size + 1$\\
        }

        Stop\\
        \label{alg:algorithm1}
    \end{algorithm}

    \subsection{Usuwanie elementów ($dequeue$).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        \If{$size < 1$}{
            Wypisz błąd o pustej kolejce.\\
        }
        \Else{
            $dequeued\_value \gets values[size - 1]$\\
            $size \gets size - 1$\\

            Wypisz $dequeued\_value$\\
        }

        Stop\\
        \label{alg:algorithm2}
    \end{algorithm}

    \subsection{Sprawdzanie elementu z przodu kolejki ($peek$).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        \If{$size < 1$}{
            Wypisz błąd o pustej kolejce.\\
        }
        \Else{
            Wypisz $values[size - 1]$\\
        }

        Stop\\
        \label{alg:algorithm3}
    \end{algorithm}
