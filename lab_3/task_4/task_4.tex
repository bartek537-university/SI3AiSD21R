    \section{Lista jednokierunkowa}

    \subsection{Dodawanie elementów ($insert$).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        Wczytaj $value$ oraz $index$\\

        \If{nie $(0 \leq index \leq size)$}{
            Wypisz błąd o niedozwolonym indeksie\\
        }
        \Else{
            $node \gets new\ Node(value)$\\

            \If{$index = 0$}{
                $node.next \gets root$\\
                $root \gets node$\\
            }
            \Else{
                $current \gets root$\\
                \For{$i \gets 0$ do $index - 1$}{
                    $current \gets current.next$\\
                }
                $node.next \gets current.next$\\
                $current.next \gets node$
            }

            $size \gets size + 1$\\
        }

        Stop\\
        \label{alg:algorithm1}
    \end{algorithm}

    \subsection{Usuwanie elementów ($remove$).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        Wczytaj $index$\\

        \If{nie $(0 \leq index < size)$}{
            Wypisz błąd o niedozwolonym indeksie\\
        }
        \Else{
            \If{$index = 0$}{
                $node \gets root$\\
                $root \gets node.next$\\
            }
            \Else{
                $current \gets root$\\
                \For{$i \gets 0$ do $index - 1$}{
                    $current \gets current.next$\\
                }
                $node \gets current.next$\\
                $current.next \gets node.next$\\
            }

            $size \gets size - 1$\\
            Wypisz $node.value$\\
        }
        Stop\\
        \label{alg:algorithm2}
    \end{algorithm}

    \subsection{Wyszukiwanie ($index$).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        Wczytaj $value$\\
        $current \gets root$\\

        \For{$i \gets 0$ do $size$}{
            \If{$current.value = value$}{
                Wypisz $i$\\
                Stop\\
            }
            $current \gets current.next$\\
        }
        Wypisz $-1$\\
        Stop\\
        \label{alg:algorithm3}
    \end{algorithm}
