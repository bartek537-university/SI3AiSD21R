    \section{Kolejka}

    \subsection{Dodawanie elementów ($enqueue$).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        Wczytaj $value$\\

        \If{$size \geq capacity$}{
            Wypisz błąd o przepełnieniu bufora.\\
        }
        \Else{
            $insert\_index \gets (start\_index + size)$ \textbf{mod} $capacity$\\
            $buffer[insert\_index] \gets value$\\
            $size \gets size + 1$
        }

        Stop\\
        \label{alg:algorithm1}
    \end{algorithm}

    \subsection{Usuwanie elementów ($dequeue$).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        \If{$size < 1$}{
            Wypisz błąd o pustej kolejce.\\
        }
        \Else{
            $dequeued\_value \gets buffer[start\_index]$\\

            $start\_index \gets (start\_index + 1)$ \textbf{mod} $capacity$\\
            $size \gets size - 1$\\

            Wypisz $dequeued\_value$\\
        }

        Stop\\
        \label{alg:algorithm2}
    \end{algorithm}

    \subsection{Sprawdzanie elementu z przodu kolejki ($peek$).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        \If{$size < 1$}{
            Wypisz błąd o pustej kolejce.\\
        }
        \Else{
            Wypisz $buffer[start\_index]$\\
        }

        Stop\\
        \label{alg:algorithm3}
    \end{algorithm}
