    \section{Sortowanie przez wstawianie (insertion sort).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\
        Wczytaj listę $values$\\
        \For{i $\gets$ 1 \KwTo \Length $values$}{
            $current \gets values[i]$\\
            $j \gets i$\\
            \While{$j > 0$ oraz $values[j-1] > current$}{
                $values[j] \gets values[j-1]$\\
                $j \gets j - 1$\\
            }
            $values[j] \gets current$\\
        }
        Wypisz listę $values$\\
        Stop\\
        \label{alg:algorithm}
    \end{algorithm}
