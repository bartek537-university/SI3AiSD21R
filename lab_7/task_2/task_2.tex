    \section{Sortowanie bąbelkowe (bubble sort).}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\
        Wczytaj listę $values$\\
        \For{i $\gets$ 0 \KwTo \Length $values$ - 1}{
            $any\_swaps\_occurred \gets fałsz$\\
            \For{j $\gets$ 0 \KwTo \Length $values - i - 1$}{
                \If{$values[j] > values[j+1]$}{
                    $temp \gets values[j+1]$\\
                    $values[j+1] \gets values[j]$\\
                    $values[j] \gets temp$\\
                    $any\_swaps\_occurred \gets prawda$\\
                }
            }
            \If{$!any\_swaps\_occurred$}{
                \Break
            }
        }
        Wypisz listę $values$\\
        Stop\\
        \label{alg:algorithm}
    \end{algorithm}
