    \section{Wyszukiwanie wzorca. Algorytm naiwny.}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        Wczytaj $text$ oraz $pattern$\\

        $text\_length \gets$ \Length $text$\\
        $pattern\_length \gets$ \Length $pattern$\\

        $matches \gets$ pusta tablica\\
        $i\_max \gets text\_length - pattern\_length + 1$\\

        \For{$i \gets 0$ \KwTo $i\_max$}{
            $j \gets 0$\\

            \While{$j < pattern\_length$}{
                \If{$text[i+j] \neq pattern[j]$}{
                    \Break\\
                }
                $j \gets j + 1$\\
            }

            \If{$j = pattern\_length$}{
                Wstaw wartość $i$ do listy $matches$\\
            }
        }

        Wypisz listę $matches$\\
        Stop\\
        \label{alg:algorithm1}
    \end{algorithm}
