    \section{NWD. Niezoptymalizowany algorytm Euklidesa.}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\
        Wczytaj współczynniki \textbf{a} i \textbf{b}\\
        \While{a $\neq$ b}{
            \If{a > b}{
                a $\gets$ a - b
            }
            \Else{
                b $\gets$ b - a
            }
        }
        Wypisz \textbf{a}\\
        Stop\\

        \label{alg:algorithm1}
    \end{algorithm}

    \section{NWD. Zoptymalizowany algorytm Euklidesa.}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\
        Wczytaj współczynniki \textbf{a} i \textbf{b}\\
        \While{b $\neq$ 0}{
            temp $\gets$ b\\
            b $\gets$ a $\mod$ b\\
            a $\gets$ temp\\
        }
        Wypisz \textbf{a}\\
        Stop\\

        \label{alg:algorithm2}
    \end{algorithm}

    \section{NWW. Zoptymalizowany algorytm Euklidesa.}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\
        Wczytaj współczynniki \textbf{a} i \textbf{b}\\
        ab\_product $\gets$ a $*$ b\\
        \While{b $\neq$ 0}{
            temp $\gets$ b\\
            b $\gets$ a $\mod$ b\\
            a $\gets$ temp\\
        }
        nwd $\gets$ $\frac{ab\_product}{a}$\\
        Wypisz \textbf{nwd}\\
        Stop\\

        \label{alg:algorithm3}
    \end{algorithm}
