    \section{Generator dodający z przeniesieniem.}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\
        Wczytaj \textbf{n}, współczynniki \textbf{m}, \textbf{j}, \textbf{k} i wartości początkowe \textbf{x}\\
        l $\gets$ długość(x)\\
        c $\gets$ 0\\
        \For{i $\gets$ 1 \KwTo n}{
            x1 $\gets$ x[(i-j+l) mod l]\\
            x2 $\gets$ x[(i-k+l) mod l]\\

            lhs $\gets$ x1 $+$ x2 $+$ c\\
            result $\gets$ lhs mod m\\

            x[i mod l] $\gets$ result\\
            \If{lhs $\geq$ m}{
                c $\gets$ 1\\
            }
            \Else{
                c $\gets$ 0\\
            }

            Wypisz \textbf{result}\\
        }
        Stop\\

        \label{alg:algorithm}
    \end{algorithm}
