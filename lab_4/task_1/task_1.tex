    \section{Szyfr Cezara - szyfrowanie.}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        Wczytaj wiadomość $message$, przesunięcie $k$ oraz alfabet $alphabet$.\\

        $a \gets$ \Length $alphabet$\\
        $result \gets ""$\\

        \For{$i \gets 0$ \KwTo \Length $message$}{
            $letter\_index \gets$ pozycja litery $message[i]$ w $alphabet$\\
            $result \gets result + alphabet[(letter\_index+k)\ \textbf{div}\ a]$\\
        }

        Wypisz $result$\\
        Stop\\
        \label{alg:algorithm1}
    \end{algorithm}

        }
