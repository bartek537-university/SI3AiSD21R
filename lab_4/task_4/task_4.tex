    \section{Algorytm Floyda-Warshalla}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        Wczytaj listę krawędzi $edges$ jako \textbf{trójki} $source$, $target$, $distance$\\

        $size \gets ($największa wartość z kolumn $edges[0]$ i $edges[1]) + 1$\\
        $distances \gets$ tablica rozmiaru $size \times size$ wypełniona $\infty$\\

        \For{$i \gets 0$ do $size$}{
            $source$, $destination$, $distance \gets edges[i]$\\
            $distances[source][destination] \gets distance$\\
        }

        \For{$i \gets 0$ do $size$}{
            $distances[i][i] \gets 0$\\
        }

        \For{$k \gets 0$ do $size$}{
            \For{$y \gets 0$ do $size$}{
                \For{$x \gets 0$ do $size$}{
                    $calculated\_distance \gets distances[y][k] + distances[k][x]$\\

                    \If{$calculated\_distance < distances[y][x]$}{
                        $distances[y][x] \gets calculated\_distance$\\
                    }
                }
            }
        }

        Wypisz $distances$\\

        Stop\\
        \label{alg:algorithm1}
    \end{algorithm}
