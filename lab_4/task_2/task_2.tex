    \section{Algorytm Dijkstry}

    \begin{algorithm}[H]
        \SetAlgoLined
        Start\\

        Wczytaj graf skierowany $graph$ w postaci macierzowej, $source$ oraz $destination$\\

        $vertex\_count \gets$ \textbf{ilość wierzchołków} $graph$\\

        $weights \gets$ tablica długości $vertex\_count$\\
        $history \gets$ tablica długości $vertex\_count$\\

        \For{$i \gets 0$ do $vertex\_count$}{
            $weights[i] \gets \infty$\\
            $history[i] \gets brak$\\
        }

        $pending\_vertices \gets$ set pojemności $vertex\_count$\\

        \For{$i \gets 0$ do $vertex\_count$}{
            Dodaj $i$ do $pending\_vertices$\\
        }

        $weights[source] \gets 0$\\

        \While{\Length $pending\_vertices > 0$}{
            $vertex \gets$ element z $pending\_vertices$ o najmniejszej wartości

            \If{$vertex = destination$}{
                \Break
            }

            $weights[vertex]$\\
            Usuń wartość $vertex$ z $pending\_vertices$\\

            \For{$neighbor \gets 0$ do $vertex\_count$}{
                $distance \gets graph[vertex][neighbor]$\\

                \If{$distance \leq 0$}{
                    \Continue
                }

                $calculated\_weight \gets weights[vertex] + distance$\\

                \If{$calculated\_weight < weights[neighbor]$}{
                    $weights[neighbor] \gets calculated\_weight$\\
                    $history[neighbor] \gets vertex$\\
                }
            }
        }

        $path \gets$ pusta lista\\
        $vertex \gets destination$\\

        \While{$vertex \neq -1$}{
            Wstaw wartość $vertex$ na pozycji $0$ do $path$\\
            $vertex \gets history[vertex]$\\
        }

        Wypisz ścieżkę $path$ oraz sumaryczny dystans $weights[destination]$\\

        Stop\\
        \label{alg:algorithm1}
    \end{algorithm}
